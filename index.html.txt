  <title>Square of a Number</title>
  <script>
    // 1. Function Declaration
    function squareDeclaration(num) {
      return num * num;
    }

    // 2. Function Expression
    const squareExpression = function(num) {
      return num * num;
    };

    // 3. Arrow Function
    const squareArrow = num => num * num;

    // Test the functions
    window.onload = () => {
      const num = 5;
      console.log("Function Declaration:", squareDeclaration(num));
      console.log("Function Expression:", squareExpression(num));
      console.log("Arrow Function:", squareArrow(num));
    };
  </script>
